 47def mask_data_to_segmentation(masks, shape, with_background):
 48    """Convert the output of the automatic mask generation to an instance segmentation."""
 49
 50    masks = sorted(masks, key=(lambda x: x["area"]), reverse=True)
 51    segmentation = np.zeros(shape[:2], dtype="uint32")
 52
 53    for seg_id, mask in enumerate(masks, 1):
 54        segmentation[mask["segmentation"]] = seg_id
 55
 56    if with_background:
 57        seg_ids, sizes = np.unique(segmentation, return_counts=True)
 58        bg_id = seg_ids[np.argmax(sizes)]
 59        if bg_id != 0:
 60            segmentation[segmentation == bg_id] = 0
 61        vigra.analysis.relabelConsecutive(segmentation, out=segmentation)
 62
 63    return segmentation

157    def _postprocess_small_regions(self, mask_data, min_area, nms_thresh):
158
159        if len(mask_data["rles"]) == 0:
160            return mask_data
161
162        # filter small disconnected regions and holes
163        new_masks = []
164        scores = []
165        for rle in mask_data["rles"]:
166            mask = amg_utils.rle_to_mask(rle)
167
168            mask, changed = amg_utils.remove_small_regions(mask, min_area, mode="holes")
169            unchanged = not changed
170            mask, changed = amg_utils.remove_small_regions(mask, min_area, mode="islands")
171            unchanged = unchanged and not changed
172
173            new_masks.append(torch.as_tensor(mask).unsqueeze(0))
174            # give score=0 to changed masks and score=1 to unchanged masks
175            # so NMS will prefer ones that didn't need postprocessing
176            scores.append(float(unchanged))
177
178        # recalculate boxes and remove any new duplicates
179        masks = torch.cat(new_masks, dim=0)
180        boxes = amg_utils.batched_mask_to_box(masks)
181        keep_by_nms = batched_nms(
182            boxes.float(),
183            torch.as_tensor(scores),
184            torch.zeros_like(boxes[:, 0]),  # categories
185            iou_threshold=nms_thresh,
186        )
187
188        # only recalculate RLEs for masks that have changed
189        for i_mask in keep_by_nms:
190            if scores[i_mask] == 0.0:
191                mask_torch = masks[i_mask].unsqueeze(0)
192                mask_data["rles"][i_mask] = amg_utils.mask_to_rle_pytorch(mask_torch)[0]
193                mask_data["boxes"][i_mask] = boxes[i_mask]  # update res directly
194        mask_data.filter(keep_by_nms)
195
196        return mask_data

198    def _postprocess_masks(self, mask_data, min_mask_region_area, box_nms_thresh, crop_nms_thresh, output_mode):
199        # filter small disconnected regions and holes in masks

202                mask_data,

204                max(box_nms_thresh, crop_nms_thresh),

207        # encode masks
208        if output_mode == "coco_rle":
209            mask_data["segmentations"] = [amg_utils.coco_encode_rle(rle) for rle in mask_data["rles"]]
210        elif output_mode == "binary_mask":
211            mask_data["segmentations"] = [amg_utils.rle_to_mask(rle) for rle in mask_data["rles"]]
212        else:
213            mask_data["segmentations"] = mask_data["rles"]

216        curr_anns = []
217        for idx in range(len(mask_data["segmentations"])):
218            ann = {
219                "segmentation": mask_data["segmentations"][idx],
220                "area": amg_utils.area_from_rle(mask_data["rles"][idx]),
221                "bbox": amg_utils.box_xyxy_to_xywh(mask_data["boxes"][idx]).tolist(),

326            data.cat(batch_data)
327            del batch_data

450    def _compute_initial_segmentation(self):
451
452        embeddings = self.predictor.get_image_embedding().squeeze().cpu().numpy()
453        assert embeddings.shape == (256, 64, 64), f"{embeddings.shape}"
454
455        initial_segmentation = embed.segment_embeddings_mws(
456            embeddings, distance_type=self.distance_type, offsets=self.offsets, bias=self.bias,
457        ).astype("uint32")
458        assert initial_segmentation.shape == (64, 64), f"{initial_segmentation.shape}"
459
460        # filter out small initial objects
461        if self.min_initial_size > 0:
462            seg_ids, sizes = np.unique(initial_segmentation, return_counts=True)
463            initial_segmentation[np.isin(initial_segmentation, seg_ids[sizes < self.min_initial_size])] = 0
464
465        # resize to 256 x 256, which is the mask input expected by SAM
466        initial_segmentation = resize(
467            initial_segmentation, (256, 256), order=0, preserve_range=True, anti_aliasing=False
468        ).astype(initial_segmentation.dtype)

487            data = amg_utils.MaskData(
488                masks=torch.from_numpy(masks),
489                iou_preds=torch.from_numpy(iou_preds),
490                seg_id=torch.from_numpy(np.full(len(masks), seg_id, dtype="int64")),
491            )
492            del masks
493            mask_data.cat(data)
494
495        return mask_data

899        scores = intersection_over_union(overlap)
900        # remove ignore_label (remapped to continuous object_ids)
901        if ignore_idx[0] is not None:

913        else:

915            costs = -(scores >= iou_threshold).astype(float) - scores / (2*n_matched)
916            seg_ind, prev_ind = linear_sum_assignment(costs)

919            match_ok = scores[seg_ind, prev_ind] >= iou_threshold
920
921            id_updates = {0: 0}
922            matched_ids, matched_prev = seg_ids[seg_ind[match_ok]], prev_ids[prev_ind[match_ok]]
923            id_updates.update(
924                {seg_id: prev_id for seg_id, prev_id in zip(matched_ids, matched_prev) if seg_id != 0}
925            )
926
927            unmatched_ids = np.setdiff1d(seg_ids, np.concatenate([np.zeros(1, dtype=matched_ids.dtype), matched_ids]))
928            id_updates.update({seg_id: max_id + i for i, seg_id in enumerate(unmatched_ids, 1)})

945        this_ids = np.unique(seg)[1:]
